#####################################################
# Bandage consumable macros

!define local ABILITY_BANDAGE_TICK
{
	string: @_string
	icon: @_icon
	states: [ default in_combat ]
	flags: [ always_in_range always_in_line_of_sight ]	
	direct_effect heal: { base: @_heal_tick }	
}

!define local AURA_BANDAGE
{
	string: @_string
	icon: @_icon
	type: buff
	flags: [ channeled ]
	duration: 50
	aura_effect repeat: 
	{ 
		update_interval: 10
		update_count: 5
		ability: @_ability_tick
	}
}

!define local ABILITY_BANDAGE	
{
	string: @_string
	description: @_description
	flags: [ target_other target_self target_friendly ]
	icon: @_icon
	consume_items: { consumable_linen_bandage: 1 }
	states: [ default in_combat ]	
	direct_effect apply_aura: { aura: @_aura }
}

!define local ITEM_BANDAGE
{
	item_level: @_level 
	required_level: @_level
	icon: @_icon
	root: { name: @_string }
	stack: 20	
	use_ability: @_ability
}

!define local ABILITY_CRAFT_BANDAGE
{
	string: @_string
	description: @_craft_description
	icon: @_icon
	cast_time: 20
	flags: [ crafting ]	
	states: [ default in_combat ]
	consume_items: @_materials
	produce_items: @_item
	required_profession<@_skill>: first_aid 
}

ability consumable_linen_bandage:	!ABILITY_BANDAGE { _string: "Linen Bandage" _description: "Restores 25 health over 5 seconds." _icon: icon_linen_bandage _aura: consumable_linen_bandage }
item consumable_linen_bandage: !ITEM_BANDAGE { _string: "Linen Bandage" _level: 1 _icon: icon_linen_bandage _ability: consumable_linen_bandage }
ability craft_linen_bandage: !ABILITY_CRAFT_BANDAGE { _string: "Linen Bandage" _craft_description: "Craft a Linen Bandage." _icon: icon_linen_bandage _materials: { item_linen_cloth: 1 } _item: { consumable_linen_bandage: 1 } _skill: 0 }
